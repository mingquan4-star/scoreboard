📋 點此展開初始化 SQL（貼到 Supabase SQL Editor 執行） 
-- 1. 選手表
create table if not exists players (
  id bigint primary key generated always as identity,
  name text not null,
  grp text not null,
  created_at timestamptz default now()
);

-- 2. 場次表
create table if not exists matches (
  id bigint primary key generated always as identity,
  p1 bigint references players(id) on delete cascade,
  p2 bigint references players(id) on delete cascade,
  grp text,
  court text,
  status text default 'pending',
  score_a int default 0,
  score_b int default 0,
  winner bigint references players(id),
  sort_order int default 0,
  created_at timestamptz default now()
);

-- 3. 密碼設定表（單列）
create table if not exists app_config (
  id int primary key default 1,
  passwords jsonb default '{
    "admin":"admin123",
    "courts":{"1號場":"court1","2號場":"court2","3號場":"court3","4號場":"court4"}
  }'::jsonb
);
insert into app_config (id) values (1) on conflict do nothing;

-- 4. 開啟 Realtime（在 Supabase Dashboard → Database → Replication 勾選 players 與 matches）
-- 或執行：
alter publication supabase_realtime add table players;
alter publication supabase_realtime add table matches;

-- 5. Row Level Security（設定公開讀、寫需 anon key）
alter table players enable row level security;
alter table matches enable row level security;
alter table app_config enable row level security;
create policy "allow all" on players for all using (true) with check (true);
create policy "allow all" on matches for all using (true) with check (true);
create policy "allow all" on app_config for all using (true) with check (true);
